75      /**
76       * Factory method to wrap a JDOM node with a wrapper that implements the Saxon
77       * NodeInfo interface.
78       *
79       * @param node       The JDOM node
80       * @param docWrapper The wrapper for the Document containing this node
81       * @param parent     The wrapper for the parent of the JDOM node
82       * @param index      The position of this node relative to its siblings
83       * @return The new wrapper for the supplied node
84       */
85  
86      protected static JDOM2NodeWrapper makeWrapper(Object node, JDOM2DocumentWrapper docWrapper,
87                                             JDOM2NodeWrapper parent, int index) {
88          JDOM2NodeWrapper wrapper;
89          if (node instanceof Document) {
90              wrapper = (JDOM2NodeWrapper)docWrapper.getRootNode();
91              if (wrapper == null) {
92                  wrapper = new JDOM2NodeWrapper(node, parent, index);
93                  wrapper.nodeKind = Type.DOCUMENT;
94              }
95          } else if (node instanceof Element) {
96              wrapper = new JDOM2NodeWrapper(node, parent, index);
97              wrapper.nodeKind = Type.ELEMENT;
98          } else if (node instanceof Attribute) {
99              wrapper = new JDOM2NodeWrapper(node, parent, index);
100             wrapper.nodeKind = Type.ATTRIBUTE;
101         } else if (node instanceof String || node instanceof Text) {
102             wrapper = new JDOM2NodeWrapper(node, parent, index);
103             wrapper.nodeKind = Type.TEXT;
104         } else if (node instanceof Comment) {
105             wrapper = new JDOM2NodeWrapper(node, parent, index);
106             wrapper.nodeKind = Type.COMMENT;
107         } else if (node instanceof ProcessingInstruction) {
108             wrapper = new JDOM2NodeWrapper(node, parent, index);
109             wrapper.nodeKind = Type.PROCESSING_INSTRUCTION;
110         } else if (node instanceof Namespace) {
111             throw new IllegalArgumentException("Cannot wrap JDOM namespace objects");
112         } else {
113             throw new IllegalArgumentException("Bad node type in JDOM! " + node.getClass() + " instance " + node.toString());
114         }
115         wrapper.treeInfo = docWrapper;
116         return wrapper;
117     }

203     /**
204      * Supporting method to get the string value of a node
205      *
206      * @param node the JDOM node
207      * @return the XPath string value of the node
208      */
209 
210     private static String getStringValue(Object node) {
211         if (node instanceof Document) {
212             List children1 = ((Document) node).getContent();
213             FastStringBuffer sb1 = new FastStringBuffer(FastStringBuffer.C256);
214             expandStringValue(children1, sb1);
215             return sb1.toString();
216         } else if (node instanceof Element) {
217             return ((Element) node).getValue();
218         } else if (node instanceof Attribute) {
219             return ((Attribute) node).getValue();
220         } else if (node instanceof Text) {
221             return ((Text) node).getText();
222         } else if (node instanceof String) {
223             return (String) node;
224         } else if (node instanceof Comment) {
225             return ((Comment) node).getText();
226         } else if (node instanceof ProcessingInstruction) {
227             return ((ProcessingInstruction) node).getData();
228         } else if (node instanceof Namespace) {
229             return ((Namespace) node).getURI();
230         } else {
231             return "";
232         }
233     }

235     /**
236      * Get the string values of all the nodes in a list, concatenating the values into
237      * a supplied string buffer
238      *
239      * @param list the list containing the nodes
240      * @param sb   the StringBuffer to contain the result
241      */
242     private static void expandStringValue(List list, FastStringBuffer sb) {
243         for (Object obj : list) {
244             if (obj instanceof Element) {
245                 sb.append(((Element) obj).getValue());
246             } else if (obj instanceof Text) {
247                 sb.append(((Text) obj).getText());
248             } else if (obj instanceof EntityRef) {
249                 throw new IllegalStateException("Unexpanded entity in JDOM tree");
250             } else if (obj instanceof DocType) {
251                 // do nothing: can happen in JDOM beta 10
252             } else {
253                 throw new AssertionError("Unknown JDOM node type");
254             }
255         }
256     }

348     public NodeInfo getParent() {
349         if (parent == null) {
350             if (node instanceof Element) {
351                 if (((Element) node).isRootElement()) {
352                     parent = makeWrapper(((Element) node).getDocument(), getTreeInfo());
353                 } else {
354                     parent = makeWrapper(((Element) node).getParent(), getTreeInfo());
355                 }
356             } else if (node instanceof Text) {
357                 parent = makeWrapper(((Text) node).getParent(), getTreeInfo());
358             } else if (node instanceof Comment) {
359                 parent = makeWrapper(((Comment) node).getParent(), getTreeInfo());
360             } else if (node instanceof ProcessingInstruction) {
361                 parent = makeWrapper(((ProcessingInstruction) node).getParent(), getTreeInfo());
362             } else if (node instanceof Attribute) {
363                 parent = makeWrapper(((Attribute) node).getParent(), getTreeInfo());
364             } else if (node instanceof Document) {
365                 parent = null;
366             } else if (node instanceof Namespace) {
367                 throw new UnsupportedOperationException("Cannot find parent of JDOM namespace node");
368             } else {
369                 throw new IllegalStateException("Unknown JDOM node type " + node.getClass());
370             }
371         }
372         return parent;

375     /**
376      * Get the index position of this node among its siblings (starting from 0)
377      * In the case of a text node that maps to several adjacent siblings in the JDOM,
378      * the numbering actually refers to the position of the underlying JDOM nodes;
379      * thus the sibling position for the text node is that of the first JDOM node
380      * to which it relates, and the numbering of subsequent XPath nodes is not necessarily
381      * consecutive.
382      */
383 
384     public int getSiblingPosition() {
385         if (index == -1) {
386             int ix = 0;
387             getParent();
388             if (parent == null) {
389                 return 0;
390             }
391             AxisIterator iter;
392             switch (nodeKind) {
393                 case Type.ELEMENT:
394                 case Type.TEXT:
395                 case Type.COMMENT:
396                 case Type.PROCESSING_INSTRUCTION:
397                     iter = parent.iterateAxis(AxisInfo.CHILD);
398                     break;
399                 case Type.ATTRIBUTE:
400                     iter = parent.iterateAxis(AxisInfo.ATTRIBUTE);
401                     break;
402                 case Type.NAMESPACE:
403                     iter = parent.iterateAxis(AxisInfo.NAMESPACE);
404                     break;
405                 default:
406                     index = 0;
407                     return index;
408             }
409             while (true) {
410                 NodeInfo n = iter.next();
411                 if (n == null) {
412                     break;
413                 }
414                 if (n.isSameNodeInfo(this)) {
415                     index = ix;
416                     return index;
417                 }
418                 if (((JDOM2NodeWrapper) n).node instanceof List) {
419                     ix += ((List) ((JDOM2NodeWrapper) n).node).size();
420                 } else {
421                     ix++;
422                 }
423             }
424             throw new IllegalStateException("JDOM node not linked to parent node");
425         }
426         return index;
427     }

464     @Override
465     protected AxisIterator iterateDescendants(NodeTest nodeTest, boolean includeSelf) {
466         Iterator<? extends Content> descendants;
467         if (nodeTest.getUType() == UType.ELEMENT) {
468             // only select element nodes
469             descendants = ((Parent) node).getDescendants(new ElementFilter());
470         } else {
471             descendants = ((Parent) node).getDescendants();
472         }
473         NodeWrappingFunction wrappingFunct = new NodeWrappingFunction<Content, NodeInfo>() {
474             public NodeInfo wrap(Content node) {
475                 return makeWrapper(node, getTreeInfo());
476             }
477         };
478         AxisIterator wrappedDescendants = new DescendantWrappingIterator(descendants, wrappingFunct);
479 
480         if (includeSelf && nodeTest.matchesNode(this)) {
481             wrappedDescendants = new PrependIterator(this, wrappedDescendants);
482         }
483 
484         if (nodeTest instanceof AnyNodeTest || (nodeTest instanceof NodeKindTest && ((NodeKindTest) nodeTest).getNodeKind() == Type.ELEMENT)) {
485             return wrappedDescendants;
486         } else {
487             return new Navigator.AxisFilter(wrappedDescendants, nodeTest);
488         }
489     }

578     /**
579      * Get all namespace undeclarations and undeclarations defined on this element.
580      *
581      * @param buffer If this is non-null, and the result array fits in this buffer, then the result
582      *               may overwrite the contents of this array, to avoid the cost of allocating a new array on the heap.
583      * @return An array of integers representing the namespace declarations and undeclarations present on
584      *         this element. For a node other than an element, return null. Otherwise, the returned array is a
585      *         sequence of namespace codes, whose meaning may be interpreted by reference to the name pool. The
586      *         top half word of each namespace code represents the prefix, the bottom half represents the URI.
587      *         If the bottom half is zero, then this is a namespace undeclaration rather than a declaration.
588      *         The XML namespace is never included in the list. If the supplied array is larger than required,
589      *         then the first unused entry will be set to -1.
590      *         <p/>
591      *         <p>For a node other than an element, the method returns null.</p>
592      */
593 
594     public NamespaceBinding[] getDeclaredNamespaces(NamespaceBinding[] buffer) {
595         if (node instanceof Element) {
596             Element elem = (Element) node;
597             List addl = elem.getAdditionalNamespaces();
598             int size = addl.size() + 1;
599             NamespaceBinding[] result = (buffer == null || size > buffer.length ? new NamespaceBinding[size] : buffer);
600             NamePool pool = getNamePool();
601             Namespace ns = elem.getNamespace();
602             String prefix = ns.getPrefix();
603             String uri = ns.getURI();
604             result[0] = new NamespaceBinding(prefix, uri);
605             int i = 1;
606             if (!addl.isEmpty()) {
607                 for (Object anAddl : addl) {
608                     ns = (Namespace) anAddl;
609                     result[i++] = new NamespaceBinding(ns.getPrefix(), ns.getURI());
610                 }
611             }
612             if (size < result.length) {
613                 result[size] = null;
614             }
615             return result;
616         } else {
617             return null;
618         }
619     }

678     /**
679      * The class ChildEnumeration handles not only the child axis, but also the
680      * following-sibling and preceding-sibling axes. It can also iterate the children
681      * of the start node in reverse order, something that is needed to support the
682      * preceding and preceding-or-ancestor axes (the latter being used by xsl:number)
683      */
684 
685     private final class ChildEnumeration extends AxisIteratorImpl {
686 
687         private JDOM2NodeWrapper start;
688         private JDOM2NodeWrapper commonParent;
689         private ListIterator children;
690         private int ix = 0;
691         private boolean downwards;  // iterate children of start node (not siblings)
692         private boolean forwards;   // iterate in document order (not reverse order)
693 
694         public ChildEnumeration(JDOM2NodeWrapper start,
695                                 boolean downwards, boolean forwards) {
696             this.start = start;
697             this.downwards = downwards;
698             this.forwards = forwards;
699 
700             if (downwards) {
701                 commonParent = start;
702             } else {
703                 commonParent = (JDOM2NodeWrapper) start.getParent();
704             }
705 
706             if (commonParent.getNodeKind() == Type.DOCUMENT) {
707                 children = ((Document) commonParent.node).getContent().listIterator();
708             } else {
709                 children = ((Element) commonParent.node).getContent().listIterator();
710             }
711 
712             if (downwards) {
713                 if (!forwards) {
714                     // backwards enumeration: go to the end
715                     while (children.hasNext()) {
716                         children.next();
717                         ix++;
718                     }
719                 }
720             } else {
721                 ix = start.getSiblingPosition();
722                 // find the start node among the list of siblings
723                 Object n = null;
724                 if (forwards) {
725                     for (int i = 0; i <= ix; i++) {
726                         n = children.next();
727                     }
728                     if (n instanceof Text) {
729                         // move to the last of a sequence of adjacent text nodes
730                         boolean atEnd = false;
731                         while (n instanceof Text) {
732                             if (children.hasNext()) {
733                                 n = children.next();
734                                 ix++;
735                             } else {
736                                 atEnd = true;
737                                 break;
738                             }
739                         }
740                         if (!atEnd) {
741                             children.previous();
742                         }
743                     } else {
744                         ix++;
745                     }
746                 } else {
747                     for (int i = 0; i < ix; i++) {
748                         children.next();
749                     }
750                     ix--;
751                 }
752             }
753         }

755         public NodeInfo next() {
756             if (forwards) {
757                 if (children.hasNext()) {
758                     Object nextChild = children.next();
759                     if (nextChild instanceof DocType) {
760                         return next();
761                     }
762                     if (nextChild instanceof EntityRef) {
763                         throw new IllegalStateException("Unexpanded entity in JDOM tree");
764                     } else if (nextChild instanceof Text) {
765                         JDOM2NodeWrapper current = makeWrapper(nextChild, getTreeInfo(), commonParent, ix++);
766                         List<Object> list = null;
767                         while (children.hasNext()) {
768                             Object n = children.next();
769                             if (n instanceof Text) {
770                                 if (list == null) {
771                                     list = new ArrayList<Object>(4);
772                                     list.add(current.node);
773                                 }
774                                 list.add(n);
775                                 ix++;
776                             } else {
777                                 // we've looked ahead too far
778                                 children.previous();
779                                 break;
780                             }
781                         }
782                         if (list != null) {
783                             current.node = list;
784                         }
785                         return current;
786                     } else {
787                         return makeWrapper(nextChild, getTreeInfo(), commonParent, ix++);
788                     }
789                 } else {
790                     return null;
791                 }
792             } else {    // backwards
793                 if (children.hasPrevious()) {
794                     Object nextChild = children.previous();
795                     if (nextChild instanceof DocType) {
796                         return next();
797                     }
798                     if (nextChild instanceof EntityRef) {
799                         throw new IllegalStateException("Unexpanded entity in JDOM tree");
800                     } else if (nextChild instanceof Text) {
801                         JDOM2NodeWrapper current = makeWrapper(nextChild, getTreeInfo(), commonParent, ix--);
802                         List<Object> list = null;
803                         while (children.hasPrevious()) {
804                             Object n = children.previous();
805                             if (n instanceof Text) {
806                                 if (list == null) {
807                                     list = new ArrayList<Object>(4);
808                                     list.add(current.node);
809                                 }
810                                 list.add(0, n);
811                                 ix--;
812                             } else {
813                                 // we've looked ahead too far
814                                 children.next();
815                                 break;
816                             }
817                         }
818                         if (list != null) {
819                             current.node = list;
820                         }
821                         return current;
822                     } else {
823                         return makeWrapper(nextChild, getTreeInfo(), commonParent, ix--);
824                     }
825                 } else {
826                     return null;
827                 }
828             }
829         }

833             return new ChildEnumeration(start, downwards, forwards);
